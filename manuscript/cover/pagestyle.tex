% Enforce consistent page style: no running headers, centered page number footer.
% This is included explicitly from build_pdf.sh to avoid relying on metadata-driven
% header-includes behavior.

\usepackage{fancyhdr}

\AtBeginDocument{%
  \pagestyle{fancy}%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
  \fancyhead[L]{}%
  \fancyhead[C]{}%
  \fancyhead[R]{}%
  \fancyfoot[C]{\thepage}%
  \setlength{\headheight}{14pt}%

  % Chapter-opening pages use the 'plain' style in the book class.
  \fancypagestyle{plain}{%
    \fancyhf{}%
    \renewcommand{\headrulewidth}{0pt}%
    \renewcommand{\footrulewidth}{0pt}%
    \fancyfoot[C]{\thepage}%
  }%
}
